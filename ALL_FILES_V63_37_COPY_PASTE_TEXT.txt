-- Omideno7 V63.37 — Safe Cloud Backup Foundation

create extension if not exists pgcrypto;

create or replace function public.om7_set_updated_at()
returns trigger
language plpgsql
security definer
as $$
begin
  new.updated_at = now();
  return new;
end;
$$;

create table if not exists public.user_profiles (
  user_id uuid primary key references auth.users(id) on delete cascade,
  email text,
  display_name text,
  language text not null default 'fa' check (language in ('fa','en','hr')),
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create table if not exists public.user_app_settings (
  user_id uuid primary key references auth.users(id) on delete cascade,
  language text not null default 'fa' check (language in ('fa','en','hr')),
  notifications_enabled boolean not null default false,
  onesignal_id text,
  timezone text,
  last_seen_at timestamptz not null default now(),
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create table if not exists public.bible365_progress (
  user_id uuid primary key references auth.users(id) on delete cascade,
  current_day integer not null default 1 check (current_day between 1 and 365),
  selected_day integer not null default 1 check (selected_day between 1 and 365),
  completed_days jsonb not null default '[]'::jsonb,
  last_read_at timestamptz,
  source text not null default 'web_app',
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create index if not exists idx_user_profiles_language on public.user_profiles(language);
create index if not exists idx_user_app_settings_language on public.user_app_settings(language);
create index if not exists idx_bible365_progress_updated_at on public.bible365_progress(updated_at);

drop trigger if exists trg_user_profiles_updated_at on public.user_profiles;
create trigger trg_user_profiles_updated_at
before update on public.user_profiles
for each row execute function public.om7_set_updated_at();

drop trigger if exists trg_user_app_settings_updated_at on public.user_app_settings;

before update on public.bible365_progress
for each row execute function public.om7_set_updated_at();

=== FILE: v63-37-cloud-backup-sync.js ===

/* Omideno7 V63.37 — Safe Cloud Backup Sync
   Purpose:
   - Adds a non-destructive cloud backup layer for signed-in users.
   - Syncs language + Bible 365 progress to Supabase when a Supabase session exists.
   - If Supabase is unavailable or tables are not installed yet, the app continues normally.
   - Does NOT change School, Daily Word, Q&A, Notifications, Bible data, or navigation.
*/
(function(){
  'use strict';

  var VERSION = 'V63.37';
  var SYNC_FLAG = 'om7_cloud_backup_v6337_last_sync';
  var INSTALL_HINT = 'om7_cloud_backup_v6337_sql_needed';
  var KEYS = {
    lang: 'lang',
    dayV6331: 'om7_bible365_selected_day_v6331',
    started: 'om7_bible365_started',
    current: 'om7_bible365_current_day',
    view: 'om7_bible365_view_day'
  };
  var state = { busy:false, lastUserId:null, warned:false, patchedStorage:false, lastSyncAt:0 };

  function log(){
    try{ console.info.apply(console, ['[Omideno7 Cloud Backup V63.37]'].concat([].slice.call(arguments))); }catch(e){}
  }
  function warn(){
    try{ console.warn.apply(console, ['[Omideno7 Cloud Backup V63.37]'].concat([].slice.call(arguments))); }catch(e){}
  }
  function safeDay(v){
    var n = parseInt(v, 10);
    if(!isFinite(n) || n < 1) return 1;
    if(n > 365) return 365;
    return n;
  }
  function getLang(){
    try{
      var l = localStorage.getItem(KEYS.lang) || document.documentElement.lang || window.currentLang || 'fa';
      return ['fa','en','hr'].indexOf(l) >= 0 ? l : 'fa';
    }catch(e){ return 'fa'; }
  }
  function setLang(l){
    if(['fa','en','hr'].indexOf(l) < 0) return;
    try{ localStorage.setItem(KEYS.lang, l); }catch(e){}
  }
  function getLocalBible365(){
    var selected = 1, current = 1;
    try{
      selected = safeDay(localStorage.getItem(KEYS.dayV6331) || localStorage.getItem(KEYS.view) || localStorage.getItem(KEYS.current) || '1');
      current = safeDay(localStorage.getItem(KEYS.current) || localStorage.getItem(KEYS.view) || localStorage.getItem(KEYS.dayV6331) || selected);
    }catch(e){}
    return { selected_day:selected, current_day:current };
  }
  function setLocalBible365(progress){
    if(!progress) return;
    var selected = safeDay(progress.selected_day || progress.current_day || 1);
    var current = safeDay(progress.current_day || selected);
    try{
      localStorage.setItem(KEYS.started, '1');
      localStorage.setItem(KEYS.dayV6331, String(selected));
      localStorage.setItem(KEYS.view, String(selected));
      localStorage.setItem(KEYS.current, String(current));
    }catch(e){}
    try{
      if(typeof window.renderBibleReader === 'function'){
        setTimeout(function(){ try{ window.renderBibleReader(); }catch(e){} }, 120);
      }
    }catch(e){}
  }
  function getTimezone(){
    try{ return Intl.DateTimeFormat().resolvedOptions().timeZone || ''; }catch(e){ return ''; }
  }
  function isNotificationsEnabled(){
    try{
      if(typeof Notification !== 'undefined' && Notification.permission === 'granted') return true;
    }catch(e){}
    return false;
  }
  function getOneSignalId(){
    try{
      if(window.OneSignal && window.OneSignal.User && typeof window.OneSignal.User.PushSubscription !== 'undefined'){
        var ps = window.OneSignal.User.PushSubscription;
        if(ps && typeof ps.id === 'string') return ps.id;
      }
    }catch(e){}
    return null;
  }

  function getClient(){
    var candidates = [
      window.omideno7Supabase,
      window.omideno7SupabaseClient,
      window.supabaseClient,
      window.SUPABASE_CLIENT,
      window.sb,
      window.om7Supabase,
      window.schoolSupabaseClient,
      window.schoolSupabase
    ];
    for(var i=0; i<candidates.length; i++){
      var c = candidates[i];
      if(c && c.auth && typeof c.from === 'function') return c;
    }
    try{
      if(window.supabase && typeof window.supabase.createClient === 'function'){
        var url = window.SUPABASE_URL || window.supabaseUrl || window.OMIDENO7_SUPABASE_URL;
        var key = window.SUPABASE_ANON_KEY || window.supabaseAnonKey || window.OMIDENO7_SUPABASE_ANON_KEY;
        if(url && key){
          window.omideno7SupabaseClient = window.supabase.createClient(url, key);
          return window.omideno7SupabaseClient;
        }
      }
    }catch(e){}
    return null;
  }

  function getUser(client){
    if(!client || !client.auth) return Promise.resolve(null);
    if(typeof client.auth.getUser === 'function'){
      return client.auth.getUser().then(function(res){ return res && res.data && res.data.user ? res.data.user : null; }).catch(function(){ return null; });
    }
    if(typeof client.auth.getSession === 'function'){
      return client.auth.getSession().then(function(res){ return res && res.data && res.data.session ? res.data.session.user : null; }).catch(function(){ return null; });
    }
    return Promise.resolve(null);
  }

  function readCloudProgress(client, userId){
    return client.from('bible365_progress')
      .select('current_day,selected_day,completed_days,updated_at')
      .eq('user_id', userId)
      .maybeSingle();
  }

  function upsertProfileAndSettings(client, user){
    var lang = getLang();
    var email = user.email || (user.user_metadata && user.user_metadata.email) || null;
    var display = (user.user_metadata && (user.user_metadata.full_name || user.user_metadata.name || user.user_metadata.display_name)) || email || null;
    var timezone = getTimezone();
    var pushId = getOneSignalId();
    var notifications = isNotificationsEnabled();
    var now = new Date().toISOString();

    return Promise.all([
      client.from('user_profiles').upsert({
        user_id: user.id,
        email: email,
        display_name: display,
        language: lang,
        updated_at: now
      }, { onConflict:'user_id' }),
      client.from('user_app_settings').upsert({
        user_id: user.id,
        language: lang,
        notifications_enabled: notifications,
        onesignal_id: pushId,
        timezone: timezone,
        last_seen_at: now,
        updated_at: now
      }, { onConflict:'user_id' })
    ]);
  }

  function upsertProgress(client, userId, local){
    var now = new Date().toISOString();
    return client.from('bible365_progress').upsert({
      user_id: userId,
      current_day: safeDay(local.current_day),
      selected_day: safeDay(local.selected_day),
      completed_days: [],
      last_read_at: now,
      source: 'web_app_v6337',
      updated_at: now
    }, { onConflict:'user_id' });
  }

  function maybeRestoreFromCloud(cloud){
    if(!cloud) return false;
    var local = getLocalBible365();
    var hasLocal = false;
    try{
      hasLocal = !!(localStorage.getItem(KEYS.dayV6331) || localStorage.getItem(KEYS.view) || localStorage.getItem(KEYS.current));
    }catch(e){}
    var cloudSelected = safeDay(cloud.selected_day || cloud.current_day || 1);
    var localSelected = safeDay(local.selected_day || 1);
    // Conservative restore: only pull cloud to device when device has no stored progress,
    // or when cloud is clearly ahead and local is still at the default day 1.
    if(!hasLocal || (localSelected === 1 && cloudSelected > 1)){
      setLocalBible365({ selected_day: cloudSelected, current_day: safeDay(cloud.current_day || cloudSelected) });
      return true;
    }
    return false;
  }

  function sync(reason){
    if(state.busy) return;
    var nowMs = Date.now();
    if(nowMs - state.lastSyncAt < 1200 && reason !== 'auth') return;
    state.lastSyncAt = nowMs;
    var client = getClient();
    if(!client){ return; }
    state.busy = true;
    getUser(client).then(function(user){
      if(!user){ state.busy = false; return; }
      state.lastUserId = user.id;
      return upsertProfileAndSettings(client, user).then(function(results){
        var profileError = results && results[0] && results[0].error;
        var settingsError = results && results[1] && results[1].error;
        if(profileError || settingsError) throw (profileError || settingsError);
        return readCloudProgress(client, user.id);
      }).then(function(res){
        if(res && res.error && res.error.code !== 'PGRST116') throw res.error;
        var restored = maybeRestoreFromCloud(res && res.data);
        var local = getLocalBible365();
        return upsertProgress(client, user.id, local).then(function(progressRes){
          if(progressRes && progressRes.error) throw progressRes.error;
          try{
            localStorage.setItem(SYNC_FLAG, new Date().toISOString());
            localStorage.removeItem(INSTALL_HINT);
          }catch(e){}
          log('sync ok', reason || 'auto', restored ? 'restored-from-cloud' : 'saved-local-to-cloud');
        });
      });
    }).catch(function(err){
      try{ localStorage.setItem(INSTALL_HINT, '1'); }catch(e){}
      if(!state.warned){
        state.warned = true;
        warn('cloud backup not active yet. Run supabase-v63-37-cloud-backup.sql once in Supabase SQL Editor.', err && (err.message || err.code || err));
      }
    }).finally(function(){ state.busy = false; });
  }

  function patchLocalStorage(){
    if(state.patchedStorage) return;
    state.patchedStorage = true;
    try{
      var originalSetItem = localStorage.setItem;
      localStorage.setItem = function(key, value){
        var result = originalSetItem.apply(this, arguments);
        if(key === KEYS.lang || key === KEYS.dayV6331 || key === KEYS.current || key === KEYS.view){
          setTimeout(function(){ sync('localStorage:' + key); }, 250);
        }
        return result;
      };
    }catch(e){}
  }

  function addMoreStatus(){
    try{
      var footer = document.querySelector('#more .footer, #mainFooter');
      if(!footer || document.getElementById('om7CloudBackupStatusV6337')) return;
      var el = document.createElement('small');
      el.id = 'om7CloudBackupStatusV6337';
      el.style.display = 'block';
      el.style.marginTop = '6px';
      el.style.opacity = '.72';
      var l = getLang();
      var txt = l === 'fa' ? 'پشتیبان ابری: فعال برای کاربران واردشده' : (l === 'hr' ? 'Cloud sigurnosna kopija: aktivna za prijavljene korisnike' : 'Cloud backup: active for signed-in users');
      el.textContent = txt;
      footer.appendChild(el);
    }catch(e){}
  }

  function installAuthListener(){
    try{
      var client = getClient();
      if(client && client.auth && typeof client.auth.onAuthStateChange === 'function' && !window.__om7CloudBackupAuthV6337){
        window.__om7CloudBackupAuthV6337 = true;
        client.auth.onAuthStateChange(function(){ setTimeout(function(){ sync('auth'); }, 350); });
      }
    }catch(e){}
  }

  function run(){
    patchLocalStorage();
    installAuthListener();
    addMoreStatus();
    sync('run');
  }

  document.addEventListener('DOMContentLoaded', run);
  window.addEventListener('load', run);
  document.addEventListener('click', function(){ setTimeout(function(){ sync('click'); addMoreStatus(); }, 500); }, true);
  document.addEventListener('change', function(){ setTimeout(function(){ sync('change'); }, 400); }, true);
  setTimeout(run, 500);
  setTimeout(run, 2000);
  setInterval(function(){ sync('interval'); }, 60000);

  window.OMIDENO7_CLOUD_BACKUP_V6337 = { sync: function(){ sync('manual'); }, getLocalBible365:getLocalBible365 };
  window.OMIDENO7_APP_VERSION = VERSION;
})();

/* Omideno7 V63.37 — final version stability lock
   Purpose: keep the More/footer version stable after older hotfix scripts try to change it.
*/
(function(){
  'use strict';
  var VERSION = 'V63.37';
  var LABEL = 'App Version: ' + VERSION;
  window.APP_VERSION = VERSION;
  window.OMIDENO7_APP_VERSION = VERSION;

  function cleanFooterVersion(){
    try{
      var footer = document.querySelector('#more .footer, #mainFooter');
      if(!footer) return;
      var nodes = footer.querySelectorAll('small.app-version,[data-app-version],#appVersion');
      for(var i=0; i<nodes.length; i++){
        if(nodes[i].id !== 'appVersion') nodes[i].remove();
      }
      var versionEl = footer.querySelector('#appVersion');
      if(!versionEl){
        versionEl = document.createElement('small');
        versionEl.id = 'appVersion';
        versionEl.className = 'app-version';
        footer.appendChild(versionEl);
      }
      versionEl.className = 'app-version';
      versionEl.setAttribute('data-app-version','v6337');
      versionEl.textContent = LABEL;

      var walker = document.createTreeWalker(footer, NodeFilter.SHOW_TEXT);
      var textNodes = [];
      while(walker.nextNode()) textNodes.push(walker.currentNode);
      textNodes.forEach(function(node){
        if(node.parentElement && node.parentElement.id === 'appVersion') return;
        if(/App Version:\s*V\d+(?:\.\d+)?/i.test(node.nodeValue || '')){
          node.nodeValue = (node.nodeValue || '').replace(/(?:\s*App Version:\s*V\d+(?:\.\d+)?\s*)+/gi, ' ');
        }
      });
    }catch(e){}
  }

  function run(){
    cleanFooterVersion();
    setTimeout(cleanFooterVersion, 150);
    setTimeout(cleanFooterVersion, 800);
  }

  document.addEventListener('DOMContentLoaded', run);
  window.addEventListener('load', run);
  document.addEventListener('click', function(){ setTimeout(cleanFooterVersion, 120); }, true);
  setInterval(cleanFooterVersion, 2500);
})();
